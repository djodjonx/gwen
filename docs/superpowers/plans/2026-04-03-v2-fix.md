  export const Physics2DKinematicSyncSystem = defineSystem(() => {
    const bodies = useQuery([RigidBody2D, Transform2D])

    onBeforeUpdate((_dt) => {
      for (const entity of bodies) {
        const body = entity.get(RigidBody2D)!
        const transform = entity.get(Transform2D)!
        transform.x = body.positionX
        transform.y = body.positionY
        transform.rotation = body.angle

- [ ] **Step 3: Implement AABB collision detection**

  In the appropriate fallback file (or create `packages/physics3d/src/fallback/collision.ts`):

  ```typescript
  /** Represents a body snapshot used for AABB overlap testing. */
  interface BodySnapshot {
    entityId: EntityId
    minX: number
    minY: number
    minZ: number
    maxX: number
    maxY: number
    maxZ: number
    isSensor: boolean
  }

  /** A pair of overlapping body IDs. */
  interface CollisionPair {
    a: EntityId
    b: EntityId
  }

  /**
   * Detects all AABB overlap pairs in O(N²).
   *
   * Acceptable for prototype use with < 200 bodies.
   * For production, use the WASM physics3d variant.
   *
   * @param bodies - Map of all registered physics bodies
   * @returns Array of overlapping body pairs
   */
  function detectAABBCollisions(bodies: Map<EntityId, BodySnapshot>): CollisionPair[] {
    const pairs: CollisionPair[] = []
    const entries = Array.from(bodies.values())
    for (let i = 0; i < entries.length; i++) {
      for (let j = i + 1; j < entries.length; j++) {
        const a = entries[i]!
        const b = entries[j]!
        if (
          a.minX <= b.maxX && a.maxX >= b.minX &&
          a.minY <= b.maxY && a.maxY >= b.minY &&
          a.minZ <= b.maxZ && a.maxZ >= b.minZ
        ) {
          pairs.push({ a: a.entityId, b: b.entityId })
        }
      }
    }
    return pairs
  }
  ```

  ```typescript
  /**
   * Generates the package.json content for a new GWEN project.
   *
   * @param name - Project name
   * @param gwenVersion - Current @gwenjs/* version to pin
   * @param modules - Optional list of additional @gwenjs/* modules
   * @returns JSON string for package.json
   *
   * @since 1.0.0
   */
  export function packageJsonTemplate(name: string, gwenVersion: string, modules: string[]): string {
    const deps: Record<string, string> = {
      '@gwenjs/app': `^${gwenVersion}`,
      '@gwenjs/core': `^${gwenVersion}`,
      '@gwenjs/renderer-canvas2d': `^${gwenVersion}`,
      '@gwenjs/input': `^${gwenVersion}`,
      ...Object.fromEntries(modules.map(m => [m, `^${gwenVersion}`])),
    }
    return JSON.stringify({
      name,
      version: '0.1.0',
      type: 'module',
      scripts: {
        dev: 'gwen dev',
        build: 'gwen build',
        postinstall: 'gwen prepare',
        lint: 'oxlint src/',
        'lint:fix': 'oxlint --fix src/',
        format: 'oxfmt src/',
        'format:check': 'oxfmt --check src/',
        typecheck: 'tsc --noEmit',
      },
      dependencies: deps,
      devDependencies: {
        '@gwenjs/cli': `^${gwenVersion}`,
        '@gwenjs/vite': `^${gwenVersion}`,
        vite: '^8.0.0',
        typescript: '^6.0.0',
        oxlint: '^0.16.0',
        oxfmt: '^0.36.0',
      },
    }, null, 2)
  }
  ```
